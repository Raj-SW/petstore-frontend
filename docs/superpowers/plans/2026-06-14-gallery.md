**Goal:** Add a full-stack Gallery (mini blog): admin-authored rich-text posts with cover + inline images, a public list + detail view, and adverts on both — mirroring the Pet Care Tips subsystem.

**Architecture:** Backend `GalleryPost` model + controller + routes + validator under `/api/gallery`, plus an admin image-upload endpoint to Cloudinary. Frontend `galleryApi`, an image-enabled RichText preset, public `GalleryPage`/`GalleryDetailPage`, and admin `AdminGallery`/`AdminGalleryForm`. Reuses the existing Advert system and design system.

**Tech Stack:** Express/Mongoose/Joi/JWT/Cloudinary (backend, Jest+supertest); React/Vite/Tailwind/Framer Motion/TipTap (frontend, Vitest+RTL).

```js
const mongoose = require('mongoose');

const CATEGORIES = ['event', 'community', 'award', 'announcement', 'behind_the_scenes'];

const galleryPostSchema = new mongoose.Schema(
  {
    title: { type: String, required: [true, 'Title is required'], trim: true, minlength: 2, maxlength: 150 },
    slug: { type: String, unique: true, lowercase: true, index: true },
    coverImage: { type: String, trim: true, default: '' },
    body: { type: String, required: [true, 'Body is required'] },
    excerpt: { type: String, trim: true, maxlength: 300, default: '' },
    category: { type: String, enum: { values: CATEGORIES, message: 'Invalid category' }, required: [true, 'Category is required'] },
    eventDate: { type: Date, required: [true, 'Event date is required'] },
    location: { type: String, trim: true, maxlength: 160, default: '' },
    tags: { type: [String], default: [] },
    featured: { type: Boolean, default: false },
    published: { type: Boolean, default: false },
    createdBy: { type: mongoose.Schema.Types.ObjectId, ref: 'User', required: true },
  },
  { timestamps: true }
);

galleryPostSchema.pre('save', function (next) {
  if (this.isModified('title')) {
    this.slug = this.title.toLowerCase().replace(/[^a-zA-Z0-9]/g, '-').replace(/-+/g, '-').replace(/^-|-$/g, '');
  }
  if (Array.isArray(this.tags)) {
    this.tags = [...new Set(this.tags.map((t) => String(t).trim().toLowerCase()).filter(Boolean))];
  }
  if (!this.excerpt && this.body) {
    this.excerpt = this.body.replace(/<[^>]*>/g, ' ').replace(/\s+/g, ' ').trim().slice(0, 160);
  }
  next();
});

galleryPostSchema.index({ category: 1, published: 1 });
galleryPostSchema.index({ featured: 1, published: 1 });
galleryPostSchema.statics.CATEGORIES = CATEGORIES;

module.exports = mongoose.models.GalleryPost || mongoose.model('GalleryPost', galleryPostSchema);
```

```js
const Joi = require('joi');
const { AppError } = require('../middlewares/errorHandler');

const CATEGORIES = ['event', 'community', 'award', 'announcement', 'behind_the_scenes'];

const normaliseTags = (v) => (Array.isArray(v) ? v : String(v).split(',')).map((t) => String(t).trim()).filter(Boolean);

const base = {
  title: Joi.string().min(2).max(150),
  body: Joi.string(),
  coverImage: Joi.string().allow('').uri({ scheme: ['http', 'https'] }),
  excerpt: Joi.string().allow('').max(300),
  category: Joi.string().valid(...CATEGORIES),
  eventDate: Joi.date().iso(),
  location: Joi.string().allow('').max(160),
  tags: Joi.alternatives().try(Joi.array().items(Joi.string()), Joi.string().allow('')),
  featured: Joi.boolean(),
  published: Joi.boolean(),
};

const createSchema = Joi.object({ ...base, title: base.title.required(), body: base.body.required(), category: base.category.required(), eventDate: base.eventDate.required() });
const updateSchema = Joi.object(base).min(1);

function run(schema) {
  return (req, res, next) => {
    if (req.body.tags !== undefined) req.body.tags = normaliseTags(req.body.tags);
    const { error, value } = schema.validate(req.body, { abortEarly: false, stripUnknown: true });
    if (error) return next(new AppError(error.details.map((d) => d.message).join(', '), 400));
    req.body = value;
    next();
  };
}

exports.validateGalleryPost = run(createSchema);
exports.validateGalleryPostUpdate = run(updateSchema);
```
